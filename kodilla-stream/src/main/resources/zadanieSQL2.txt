SELECT
    ISSUES.ID AS ISSUE_ID,
    ISSUES.SUMMARY,
    ISSUES.DESCRIPTION,
    ISSUESLISTS.NAME AS ISSUE_LIST_NAME
FROM
    ISSUES
        JOIN
    ISSUESLISTS ON ISSUES.ISSUESLIST_ID = ISSUESLISTS.ID;


SELECT
    ISSUES.ID AS ISSUE_ID,
    ISSUES.SUMMARY,
    ISSUES.DESCRIPTION,
    USERS.FIRSTNAME,
    USERS.LASTNAME
FROM
    ISSUES
        JOIN
    USERS ON ISSUES.USER_ID_ASSIGNEDTO = USERS.ID;

SELECT
    USERS.FIRSTNAME,
    USERS.LASTNAME,
    COUNT(ISSUES.ID) AS TASK_COUNT
FROM
    USERS
        JOIN
    ISSUES ON USERS.ID = ISSUES.USER_ID_ASSIGNEDTO
GROUP BY
    USERS.ID, USERS.FIRSTNAME, USERS.LASTNAME;
